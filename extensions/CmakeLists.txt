
INCLUDE_DIRECTORIES(
	${BULLET_PHYSICS_SOURCE_DIR}/src
	${BULLET_PHYSICS_SOURCE_DIR}/../
	${BULLET_PHYSICS_SOURCE_DIR}/../extensions
)

SET(Extensions_SRCS
ccDiscreteDynamicsWorld.cpp
ccKinematicCharacterController.cpp
ccCompoundShape.cpp
)

SET(Extensions_HDRS
ccDiscreteDynamicsWorld.h
ccCompoundShape.h
ccMaterial.h
ccKinematicCharacterController.h
ccOverlapFilterCallback.h
ccRayResultCallback.h
ccConvexCastCallBack.h
)

ADD_LIBRARY(Extensions ${Extensions_SRCS} ${Extensions_HDRS})
SET_TARGET_PROPERTIES(Extensions PROPERTIES VERSION 2.82)
SET_TARGET_PROPERTIES(Extensions PROPERTIES SOVERSION 2.82)

IF (INSTALL_LIBS)
	IF (NOT INTERNAL_CREATE_DISTRIBUTABLE_MSVC_PROJECTFILES)
		#FILES_MATCHING requires CMake 2.6
		IF (${CMAKE_MAJOR_VERSION}.${CMAKE_MINOR_VERSION} GREATER 2.5)
			IF (APPLE AND BUILD_SHARED_LIBS AND FRAMEWORK)
				INSTALL(TARGETS Extensions DESTINATION .)
			ELSE (APPLE AND BUILD_SHARED_LIBS AND FRAMEWORK)
				INSTALL(TARGETS Extensions
					RUNTIME DESTINATION bin
					LIBRARY DESTINATION lib${LIB_SUFFIX}
					ARCHIVE DESTINATION lib${LIB_SUFFIX})
				INSTALL(DIRECTORY ${CMAKE_CURRENT_SOURCE_DIR}
DESTINATION ${INCLUDE_INSTALL_DIR} FILES_MATCHING PATTERN "*.h"  PATTERN
".svn" EXCLUDE PATTERN "CMakeFiles" EXCLUDE)
			ENDIF (APPLE AND BUILD_SHARED_LIBS AND FRAMEWORK)
		ENDIF (${CMAKE_MAJOR_VERSION}.${CMAKE_MINOR_VERSION} GREATER 2.5)

		IF (APPLE AND BUILD_SHARED_LIBS AND FRAMEWORK)
			SET_TARGET_PROPERTIES(Extensions PROPERTIES FRAMEWORK true)
			SET_TARGET_PROPERTIES(Extensions PROPERTIES PUBLIC_HEADER "${Extensions_HDRS}")
		ENDIF (APPLE AND BUILD_SHARED_LIBS AND FRAMEWORK)
	ENDIF (NOT INTERNAL_CREATE_DISTRIBUTABLE_MSVC_PROJECTFILES)
ENDIF (INSTALL_LIBS)
